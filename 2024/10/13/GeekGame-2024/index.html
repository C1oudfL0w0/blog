import os
import zipfile

# 指定包含zip文件的目录路径
zip_dir = '.'

# 遍历指定目录下的所有文件
for root, dirs, files in os.walk(zip_dir):
    for file in files:
        if file.endswith('.zip'):
            zip_file = os.path.join(root, file)
            with zipfile.ZipFile(zip_file, 'r') as zip_ref:
                # 指定解压目录为当前zip文件所在目录
                zip_ref.extractall(root)
            print(f'解压 {zip_file} 完成')

print('所有zip文件解压完成')
