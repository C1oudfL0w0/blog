import requests

# 扫描地址
url1 = "需要扫描的地址"

# 常见网站源码备份文件名
dir1 = ['web','website','backup','www','wwwroot','temp']

# 常见网站源码备份文件后缀
final2 = ['tar','tar.gz','zip','rar','bak']

# 开始扫描
for i in dir1:
    for j in final 2:
        # 拼接备份文件名
        filename = str(i) + '.' + str(j)
        # 拼接最终url
        url = str(url1) + '/' + filename
        # 返回文件名
        print(filename + ' ',end='')
        # 返回数据包长度，根据长度判断网站备份文件
        print(len(request.get(url).text))
